    @staticmethod
    def connect():
        """
        Creates initial connect message.

        Returns
        -------
        bytes
            Connect message.
        """

        command_code = Command.CONNECT
        cs = CSConnect.securityString
        b = struct.pack("<ii", command_code, 20)
        return b + cs.zfill(20)
